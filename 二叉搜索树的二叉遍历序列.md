# 题目描述
输入一个整数数组，判断该数组是不是某二叉搜索树的后序遍历的结果。如果是则输出Yes,否则输出No。假设输入的数组的任意两个数字都互不相同。


## 个人评价
这个程序写得不好，代码逻辑非常的繁琐，有待优化（转指flag部分，还有边界条件的判断（empty部分））

```
#include <vector>

using namespace std;

class Solution {
public:
    bool VerifySquenceOfBST(vector<int> sequence) {
        if(sequence.empty())
            return false;
        int size = sequence.size();
        //if(size==1)
        //    return true;
        
        int root = sequence[size-1];
        int i = 0; int midp=0;
        bool f_SL = false;
        for(; i<size-1; i++)
        {
            if(!f_SL)
                if(sequence[i]<root)
                    midp = i+1;
                else
                    f_SL = true;
            else
                if(sequence[i]<root)
                    return false;
        }
        
        vector<int> seqLeft(sequence.begin(), sequence.begin()+midp);
        vector<int> seqRight(sequence.begin()+midp, sequence.end()-1);
        if( (seqLeft.empty() ||VerifySquenceOfBST(seqLeft)) && (seqRight.empty() || VerifySquenceOfBST(seqRight)))
            return true;
        else
            return false;
    }
};
```
